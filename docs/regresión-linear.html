13 Regresión linear

pacman::p_load(jtools, report)
model <- iris %>%
  lm(formula = Sepal.Length ~ Petal.Width)

# R base
anova(model)
summary(model)

# jtools
summ(model)
library(ggfortify)
autoplot(model)

Paquete report

report(model)

Agregar formula y R2

library(ggpmisc)
formu <- y ~ x

ggplot(iris, aes(x=Petal.Width, y=Sepal.Length))+
  geom_point() +
  geom_smooth(method="lm")+
  stat_poly_eq(formula = formu,
               aes(label = paste(..eq.label.., ..rr.label.., sep = "~~~")),
               parse = TRUE)

Paquete jtools

effect_plot(model, 
            pred=Petal.Width, 
            plot.points = TRUE, 
            interval = TRUE, 
            colors = "red")+
  theme_bw()

Paquete ggeffects

res_reg <- ggpredict(model, terms ="Petal.Width")

res_reg %>% 
  ggplot()+
  aes(x=x, y=predicted)+
  geom_line()

plot(res_reg, ci = FALSE) + 
  labs(x = "Ancho de pétalos") + 
  geom_point(data=iris, 
             aes(x=Petal.Width, y=Sepal.Length))
